    ```php
    <?php
    ```

    ### Nested Loops for Prime Detection

    ```php
    <?php

    for ($i = 0; $i < 100; $i++) {
        if (!is_numeric($i) || $i < 2) {
            $esPrimo = false;
        }

        for ($j = 2; $j < $i; $j++) {
            if ($i % $j == 0) {
                $esPrimo = false;
            }
        }

            $contarPrimos++;
        }

        $esPrimo = true;
    }
    ```

  <Tab title="Functions">
    ### Prime & Even/Odd Counter

    ```php
    <?php
    function func1()
    {
        $esPrimo = true;
        $contarPrimos = 0;
        $contarNoPrimos = 0;
        $contarPares = 0;
        $contarImpares = 0;

        for ($i = 0; $i < 100; $i++) {
            // Count even numbers
            if ($i % 2 == 0) {
                $contarPares++;
            }

            // Count odd numbers
            if ($i % 2 == 1) {
                $contarImpares++;
            }

            // Check if prime
            if (!is_numeric($i) || $i < 2) {
                $esPrimo = false;
            }

            for ($j = 2; $j < $i; $j++) {
                if ($i % $j == 0) {
                    $esPrimo = false;
                }
            }

            if ($esPrimo) {
                $contarPrimos++;
            } else {
                $contarNoPrimos++;
            }

            $esPrimo = true;
        }

        return array(
            "PRIMOS" => $contarPrimos,
            "NO PRIMOS" => $contarNoPrimos,
            "PARES" => $contarPares,
            "NO PARES" => $contarImpares
        );
    }

    // Usage
    foreach (func1() as $key => $value) {
        echo "$key = $value<br>";
    }
    ```

    ### Fibonacci Generator

    ```php
        $fibonacci = [];

        for ($i = 0; $i <= 150; $i++) {
            if ($i < 2) {
    ```

<Tabs>
  <Tab title="Connection">
    ### PDO Database Connection

    ```php
    <?php
    class Database
    {
        private $conexion;

        public function __construct()
        {
            $options = [
                PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
                PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
            ];

            $this->conexion = new PDO(
                'mysql:host=localhost;dbname=yixondb;charset=utf8',
                'root',
                '',
                $options
            );

            $this->conexion->exec('SET CHARACTER SET UTF8');
        }

        public function getConexion()
        {
            return $this->conexion;
        }

        public function closeConexion()
        {
            $this->conexion = null;
        }
    }
    ```

    ### Usage Example

    ```php
    <?php
    require_once "database.php";

    $database = new Database();
    $conexion = $database->getConexion();
    ```

  <Tab title="CRUD Operations">
    ### Create (Insert)

    ```php
    <?php
    public function save($data)
    {
        // Filter empty values
        $data = array_filter($data, function($value) {
            return $value !== null && $value !== '';
        });

        if (empty($data)) {
            return false;
        }

        // Build SQL query
        $columns = array_keys($data);
        $placeholders = array_map(fn($col) => ":$col", $columns);

        $sql = "INSERT INTO {$this->table} (";
        $sql .= implode(', ', $columns);
        $sql .= ") VALUES (";
        $sql .= implode(', ', $placeholders);
        $sql .= ")";

        // Execute
        $stmt = $this->db->prepare($sql);
        return $stmt->execute($data);
    }
    ```

    ### Read (Select)

    ```php
    <?php
    // Get all records
    public function getAll()
    {
        $stmt = $this->db->prepare("SELECT * FROM {$this->table}");
        $stmt->execute();
        return $stmt->fetchAll();
    }

    // Get single record by ID
    public function getById($id)
    {
        $stmt = $this->db->prepare(
            "SELECT * FROM {$this->table} WHERE {$this->id} = :id"
        );
        $stmt->bindParam(':id', $id);
        $stmt->execute();
        return $stmt->fetch();
    }
    ```

    ### Update

    ```php
    <?php
    public function updateById($id, $data)
    {
        // Filter empty values
        $data = array_filter($data, function ($value) {
            return $value !== null && $value !== '';
        });

        if (empty($data)) {
            return true;
        }

        // Build SET clause
        $set_clauses = [];
        foreach (array_keys($data) as $key) {
            $set_clauses[] = "{$key} = :{$key}";
        }

        $sql = "UPDATE {$this->table} SET ";
        $sql .= implode(', ', $set_clauses);
        $sql .= " WHERE {$this->id} = :id_value";

        // Execute
        $stmt = $this->db->prepare($sql);
        $bind_params = array_merge($data, ['id_value' => $id]);
        return $stmt->execute($bind_params);
    }
    ```

    ### Delete

    ```php
    <?php
    ```
    ```php
    class Cliente
    {
        private $nombre;

            $this->nombre = $nombre;

            $this->cuentas[] = $cuenta;
    ```

  <Tab title="Inheritance">
    ### Base Class

    ```php
    <?php
    class CuentaBanco
    {
        private $numeroCuenta;
        private $tipo;
        private $titular;
        private $saldo;
        private $claveCajero;
        private $transacciones = [];

        public function __construct(
            $numeroCuenta,
            $tipo,
            $titular,
            $claveCajero,
            $saldoInicial = 0
        ) {
            $this->numeroCuenta = $numeroCuenta;
            $this->tipo = $tipo;
            $this->titular = $titular;
            $this->claveCajero = $claveCajero;
            $this->saldo = $saldoInicial;
        }

        public function depositar($monto)
        {
            $this->saldo += $monto;
            $this->RegistrarTransaccion('depósito', $monto);
        }

        public function retirar($monto)
        {
            if ($monto <= $this->saldo) {
                $this->saldo -= $monto;
                $this->RegistrarTransaccion('retiro', $monto);
            } else {
                throw new Exception("Saldo insuficiente");
            }
        }
    }
    ```

## Security Best Practices

<CardGroup cols={2}>
  <Card title="Password Hashing" icon="lock">
    Always use `password_hash()` with PASSWORD_BCRYPT or PASSWORD_ARGON2ID
    ```php
    password_hash($_POST['pass'], PASSWORD_BCRYPT)
    ```
  </Card>

  <Card title="Prepared Statements" icon="shield">
    Use PDO prepared statements to prevent SQL injection
    ```php
    $stmt = $db->prepare("SELECT * WHERE id = :id");
    $stmt->bindParam(':id', $id);
    ```
  </Card>

  <Card title="Input Validation" icon="filter">
    Filter and validate all user input before processing
    ```php
    $data = array_filter($data, function($v) {
        return $value !== null && $value !== '';
    });
    ```
  </Card>

  <Card title="Error Handling" icon="triangle-exclamation">
    Use PDO error mode for proper exception handling
    ```php
    PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION
    ```
  </Card>
